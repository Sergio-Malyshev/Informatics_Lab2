### 2. Код скрипта
```
if [ $# -ne 1 ]; then
    echo "Использование: $0 <IPv4-адрес>"
    echo "Пример: $0 192.168.10.1"
    exit 1
fi

IP="$1"

# Базовая проверка формата (4 числа от 0-255 через точки)
if [[ ! $$   IP =~ ^[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}   $$ ]]; then
    echo "Ошибка: некорректный формат IPv4-адреса"
    exit 1
fi

# Разбиваем строку по точкам на четыре октета
IFS='.' read -r o1 o2 o3 o4 <<< "$IP"

# Функция: перевод десятичного числа в 8-битную двоичную строку
to_bin() {
    local n=$1
    local bin=""
    for ((i=7; i>=0; i--)); do
        if (( (n & (1 << i)) != 0 )); then
            bin="${bin}1"
        else
            bin="${bin}0"
        fi
    done
    echo "$bin"
}

# Преобразуем каждый октет
b1=$(to_bin "$o1")
b2=$(to_bin "$o2")
b3=$(to_bin "$o3")
b4=$(to_bin "$o4")

# Выводим результат
echo "$$   {b1}.   $${b2}.$$   {b3}.   $${b4}"
```
